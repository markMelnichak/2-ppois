50void Ingredient::addAmount(double v) {
51 double oldGrams = quantity.toGrams();
52 double newGrams = oldGrams + v;
53 if (oldGrams > 0.0) {
54 double factor = newGrams / oldGrams;
55 quantity.scale(factor);
56 }
57}

59void Ingredient::useAmount(double vGrams) {
60 double grams = quantity.toGrams();
61 if (vGrams > grams) {
62 throw NotEnoughIngredientException("Not enough ingredient");
63 }
64 if (grams > 0.0) {
65 double factor = (grams - vGrams) / grams;
66 quantity.scale(factor);
67 }
68}

273 : startTemp(s), targetTemp(t), duration(d), gradual(g) {}

275double TemperatureProfile::currentTemp(int elapsed) const {
276 if (!gradual || elapsed >= duration || duration <= 0) {
277 return targetTemp;
278 }
279 double ratio = static_cast<double>(elapsed) / duration;
280 return startTemp + (targetTemp - startTemp) * ratio;
281}
282
283bool TemperatureProfile::isReached(double current) const {
284 return current >= targetTemp;
285}
286
287void TemperatureProfile::reset(double s, double t, int d) {
288 startTemp = s;
289 targetTemp = t;
290 duration = d;
